cmake_minimum_required(VERSION 3.15)
project(realm_msgs)

find_package(catkin REQUIRED COMPONENTS
        roscpp
        std_msgs
        geometry_msgs
        sensor_msgs
        message_generation
)

add_message_files(
  FILES
  Frame.msg
  Georeference.msg
  GroundImageCompressed.msg
  Pinhole.msg
  CvGridMap.msg
  Depthmap.msg
  PointCloud.msg
)

add_service_files(
  FILES
  ParameterChange.srv
)

generate_messages(
  DEPENDENCIES
  std_msgs
  geometry_msgs
  sensor_msgs
)

####################
## Catkin Package ##
####################
set(FBOW_INCLUDE_DIRS "/usr/local/include")
set(FBOW_LIBS "/usr/local/lib/libfbow.so")
find_package(fbow REQUIRED)

catkin_package(
        CATKIN_DEPENDS
            roscpp
            std_msgs
            sensor_msgs
            geometry_msgs
            message_runtime
)
